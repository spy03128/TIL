### 대부분의 그리디 문제는 정렬과 동반된다.

### 회의실 배정

회의가 끝나는 시간 순으로 정렬

```python
n = int(input())
meeting = []

for i in range(n):
	s,e = map(int,input().split())
	meeting.append((s,e))
meeting.sort(key=lambda x: (x[1],x[0]))
endtime = 0
cnt = 0
for s,e in meeting:
	if s >=endtime:
		cnt+=1
		endtime = e

print(cnt)
```

```python
import heapq as hq

q=[]
n = int(input())
for _ in range(n):
    s,e = map(int, input().split())
    hq.heappush(q,(e,s))

cnt, end = 0,0
while q:
    e,s = hq.heappop(q)
    if s >= end:
        cnt+=1
        end = e

print(cnt)
```

### 씨름 선수

```python
n = int(input())
arr = []
for i in range(n):
    x,y = map(int,input().split())
    arr.append((x,y))

arr.sort(key=lambda x: (x[0],x[1]),reverse=True)

cnt = 0
largest = 0
for i in range(n):
    if arr[i][1] > largest:
        cnt+=1
        largest = arr[i][1]
print(cnt)
```

```python
n = int(input())
arr = []
for i in range(n):
    x,y = map(int,input().split())
    arr.append((x,y))

arr.sort(reverse=True)

cnt = 0
largest = 0
for x,y in arr:
    if y > largest:
        cnt+=1
        largest = y
print(cnt)
```

### 창고 정리

```python
n = int(input())
arr = list(map(int,input().split()))
num = int(input())

for i in range(num):
    arr.sort()
    arr[0]+=1
    arr[n-1]-=1
arr.sort()
print(arr[n-1]-arr[0])
```

### 침몰하는 타이타닉

```python
n,m = map(int,input().split())
arr = list(map(int,input().split()))

arr.sort()
s,e = 0, n-1
cnt = 0
while True:
    if s>e:
        break
    cnt+=1
    e-=1
    if s+e<=m:
        s+=1

print(cnt)
```

n, limit = map(int, input().split())
p = list(map(int, input().split())
p.sort()
cnt = 0

while p:
		if len(p) == 1:
				cnt+=1
				break
		if p[0] + p[-1] > limit:
