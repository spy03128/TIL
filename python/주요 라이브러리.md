### 내장함수

```python
result = sum([1,2,3,4,5])
print(result)
#15

result = min(7,3,5,2)
print(result)
#2

result = max(7,3,5,2)
print(result)
#7

result = eval("(3+5) * 7")
print(result)
#56

result = sorted([9,1,8,5,4])
print(result)
#[1,4,5,8,9]

result = sorted([9,1,8,5,4], reverse = True)
print(result)
#[9,8,5,4,1]

result = sorted([('홍길동',35), ('이순신',75), ('아무개',50)], key = lambda x: x[1], reverse = True)
print(result)
#[('이순신',75), ('아무개',50), ('홍길동',35)]

data = [9, 1, 8, 5, 4]
data.sort()
print(data)
#[1,4,5,8,9]
```
